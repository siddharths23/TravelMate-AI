import React, { useState, useEffect } from "react";
import { Link } from "react-router-dom";
import "../css/Navbar.css";

/**
 * The Navbar component displays the navigation bar with links to various parts of the application.
 * @function Navbar
 * @param {Object} props - The component's properties.
 * @param {Object} props.user - The currently logged-in user (if any).
 * @returns {React.Element} The rendered Navbar component.
 */
const Navbar = ({ user }) => {
  const [loggedIn, setLoggedIn] = useState(false);

  useEffect(() => {
    const token = localStorage.getItem("token");
    setLoggedIn(token !== null);
  }, []);

  /**
   * Handles the logout process by removing the token from local storage and reloading the page.
   * @function handleLogout
   */

  const handleLogout = () => {
    localStorage.removeItem("token");
    window.location.reload();
  };

  return (
    <nav className="navbar">
      <div className="logo-container">
        <Link to="/" className="logo-text">
          TravelMate
        </Link>
      </div>
      <div className="links-container">
        <Link to="/explore" className="nav-link">
          Explore
        </Link>
        <Link to="/profile" className="nav-link">
          Profile
        </Link>
        {loggedIn ? (
          <button onClick={handleLogout} className="nav-link">
            Logout ({user.lastName})
          </button>
        ) : (
          <Link to="/login" className="nav-link">
            Login
          </Link>
        )}
      </div>
    </nav>
  );
};

export default Navbar;
